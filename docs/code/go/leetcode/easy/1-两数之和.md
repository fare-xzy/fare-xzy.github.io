package main

import "fmt"

func twoSum(nums []int, target int) []int {
	numsMap := make(map[int]int)
	for i := 0; i < len(nums); i++ {
		result := target - nums[i]
		if _, ok := numsMap[result]; ok {
			return []int{numsMap[result], i}
		} else {
			numsMap[nums[i]] = i
		}
	}
	return []int{}
}
